57char hex2char(char halfhex) {



58    char rv;



59



60    // Test for numeric characters



61    if ((rv = halfhex - '0') <= 9 && rv >= 0) {



62        return rv;



63    }        // Otherwise check for upper-case A-F



64    else if ((rv = halfhex - 'A') <= 5 && rv >= 0) {



65        return rv + 10;



66    }        // Finally check for lower-case a-f



67    else if ((rv = halfhex - 'a') <= 5 && rv >= 0) {



68        return rv + 10;



69    }



70



71    // Otherwise return -1 as an error



72    return -1;



73}

81// then assembles full messages, verifies the checksum and

84unsigned char gpsUbloxSeparate(unsigned char* outStream) {



85    // Static variables CAREFUL



86    static unsigned char outBuf [MSIZE] = {0};



87    static unsigned char previousComplete = 1;



88    static unsigned char indexLast = 0;



89



90    // local variables



91    unsigned char i = 0;



92    unsigned char tmpChksum = 0;



93    unsigned char chsmStr_0, chsmStr_1;



94    unsigned char isValid = 0;



95    unsigned char chksumHeader = 0;



96    unsigned char tmpIndex = 0;



97    unsigned char tmpLen = getLength(uartBuffer);



98



99    // If the previous message was complete, then



100    // go over the buffer and advance until you find a dollar sign



101    if (previousComplete) {



102        while (tmpLen > 0 && peak(uartBuffer) != '$') {



103            readFront(uartBuffer);



104            tmpLen--;



105        }



106    }



107



108    // read until you find a CR or run out of bytes to read



109    while (tmpLen > 0 && peak(uartBuffer) != 13) { //CR = 13



110        outBuf[indexLast] = readFront(uartBuffer);



111        indexLast++;



112        tmpLen--;



113    }



114



115    // if we found a carriage return, then the message is complete



116    if (peak(uartBuffer) == 13) {



117        // validate the checksum



118        chsmStr_0 = hex2char(outBuf[indexLast - 2]);



119        chsmStr_1 = hex2char(outBuf[indexLast - 1]);



120        // convert the hex checksum to decimal



121        tmpChksum = (unsigned char) (chsmStr_0 * 16 + chsmStr_1);



122



123        // verify the validity



124        isValid = (tmpChksum == getChecksum(outBuf, indexLast));



125



126        // turn on the flag of complete stream



127        previousComplete = 1;



128        // set the outStream valid flag



129        outStream[MSIZE - 1] = isValid;



130



131        // save the index value for copying purposes



132        tmpIndex = indexLast;



133



134        // reset the array index



135        indexLast = 0;



136    } else { // if we did not find a CR, i.e. we ran out of bytes



137        // turn off the previousComplete flag



138        previousComplete = 0;



139    }



140



141    // If we found a complete sentence and it is valid



142    if (previousComplete && isValid) {



143        // copy the data to the outstream



144        for (i = 1; i <= tmpIndex; ++i) {



145            outStream[i] = outBuf[i - 1];



146        }



147



148        // finally compute the type of message



149        chksumHeader = getChecksum(outBuf, 6);



150



151        // based on the obtained header checksum set the type



152        switch (chksumHeader) {



153            case GGACS:



154                outStream[0] = GGAID;



155                break;



156            case RMCCS:



157                outStream[0] = RMCID;



158                break;



159            default:



160                outStream[0] = UNKID;



161                break;



162        }



163    }



164    return tmpLen;



165}

167void gpsUbloxParse(void) {



168



169    unsigned char inStream[MSIZE];



170    unsigned char bufferLen = 11;



171



172    memset(inStream, 0, MSIZE);



173



174    while (bufferLen > 10) {



175        bufferLen = gpsUbloxSeparate(inStream);



176



177        // if the sentence is valid



178        if (inStream[MSIZE - 1] == 1) {



179            // parse the data according to the header



180            switch (inStream[0]) {



181                case RMCID:



182                    parseRMC(inStream);



183                    break;



184                case GGAID:



185                    parseGGA(inStream);



186                    break;



187            }



188        }



189



190        memset(inStream, 0, MSIZE);



191    }



192



193}

195void getGpsUbloxMainData(float* data) {



196    gpsUbloxParse();



197    data[0] = INT32_1E7_TO_FLOAT(mlGpsData.lat);



198    data[1] = INT32_1E7_TO_FLOAT(mlGpsData.lon);



199    data[2] = INT32_1E3_TO_FLOAT(mlGpsData.alt);



200    data[3] = UINT16_1E2_TO_FLOAT(mlGpsData.cog);



201    data[4] = UINT16_1E2_TO_FLOAT(mlGpsData.vel);



202}

204char gpSmbl(char symbl) {



205    switch (symbl) {



206        case 'N': return 1; /*North*/



207        case 'E': return 1; /*East*/



208        case 'S': return -1; /*South*/



209        case 'W': return -1; /*West*/



210        case 'A': return 3; /*Valid or Autonomous*/



211        case 'D': return 1; /*Differential*/



212        case 'V': return 0; /*Non-Valid*/



213        case 'M': return 100; /*Meters*/



214        default: return 0; /*Non-Valid*/



215    }



216}

218// a return value of 1 means the string is done. No more tokens



219// This function is stateful, call it once with the String and then with NULL



220// similar to strtok but this will support succesive tokens like



221// "find,,,the,,commas"



222



223unsigned char myTokenizer(char* stringToTokenize, char token, char * returnToken) {



224    static char * pch;



225    static char * prevPch;



226    static char * lastByte;



227



228    // Make sure the return token is "empty"



229    memset(returnToken, 0, TOKEN_SIZE);



230



231    // get the pointer to next token if it exists 



232    // and the stringToTokenize is null



233    // Bahavior similar to strtok



234    if (stringToTokenize == NULL) {



235        pch = strchr(prevPch, token);



236    } else {



237        pch = strchr(stringToTokenize, token);



238        prevPch = stringToTokenize;



239        lastByte = stringToTokenize + strlen(stringToTokenize);



240    }



241



242    if (pch != NULL) {



243        memcpy(returnToken, prevPch, pch - prevPch);



244        prevPch = pch + 1;



245    } else {



246        memcpy(returnToken, prevPch, lastByte - prevPch);



247    }



248



249    return pch == NULL;



250}

252void parseRMC(unsigned char* stream) {



253    // declare the local vars



254    char token[TOKEN_SIZE];



255    char tmp [3] = {0, 0, '\0'}, tmp3[4] = {0, 0, 0, '\0'};



256    unsigned char chTmp = 0;



257



258



259    // initialize tokenizer, let go first token which holds the msg type



260    // token = strtok(stream, ",");



261    myTokenizer((char *)stream, ',', token);



262



263    // 1.- hhmmss.ssss



264    myTokenizer(NULL, ',', token);



265    if (strlen(token) > 5) {



266//        tmp[0] = token[0];



267//        tmp[1] = token[1];



268//        mlGpsDateTime.hour = (unsigned char) atoi(tmp);



269//        tmp[0] = token[2];



270//        tmp[1] = token[3];



271//        mlGpsDateTime.min = (unsigned char) atoi(tmp);



272//        tmp[0] = token[4];



273//        tmp[1] = token[5];



274//        mlGpsDateTime.sec = (unsigned char) atoi(tmp);



275    }



276



277    // 2.- Status of position Fix



278    myTokenizer(NULL, ',', token);



279    if (strlen(token) == 1) {



280        mlGpsData.fix_type = gpSmbl((char) token[0]);



281    }



282



283    // 3.- Latitude



284    // ddmm.mmmmmm



285    myTokenizer(NULL, ',', token);



286    if (strlen(token) > 0) {



287        // get the first two values



288        tmp[0] = token[0];



289        tmp[1] = token[1];



290        // get the degrees



291        chTmp = (unsigned char) atoi(tmp);



292        // make the degrees zero for minutes conversion



293        token[0] = '0';



294        token[1] = '0';



295        // get the float



296        mlGpsData.lat = FLOAT_TO_INT32_1E7(degMinToDeg(chTmp, atof(token)));



297



298        // 4.- Latitude Sector



299        myTokenizer(NULL, ',', token);



300        if (strlen(token) == 1) {



301            // set the sign of the float value



302            mlGpsData.lat *= gpSmbl((char) token[0]);



303        }



304    }



305



306    // 5.- Longitude



307    // dddmm.mmmmmm



308    myTokenizer(NULL, ',', token);



309    if (strlen(token) > 0) {



310        // get the first two values



311        tmp3[0] = token[0];



312        tmp3[1] = token[1];



313        tmp3[2] = token[2];



314        // get the degrees



315        chTmp = (unsigned char) atoi(tmp3);



316        // make the degrees zero for minutes conversion



317        token[0] = '0';



318        token[1] = '0';



319        token [2] = '0';



320        // get the float



321        mlGpsData.lon = FLOAT_TO_INT32_1E7(degMinToDeg(chTmp, atof(token)));



322



323        // 6.- Longitude Sector



324        myTokenizer(NULL, ',', token);



325        if (strlen(token) == 1) {



326            // set the sign of the float value



327            mlGpsData.lon *= gpSmbl((char) token[0]);



328        }



329    }



330



331    // 7.- SOG in knots but gets stored in m/s CAUTION



332    // xx.xx



333    myTokenizer(NULL, ',', token);



334    if (strlen(token) > 0) {



335        mlGpsData.vel = FLOAT_TO_UINT16_1E2((unsigned short) (atof(token) * KTS2MPS));



336    }



337



338    // 8.- COG in degrees



339    // xxx.xxx



340    myTokenizer(NULL, ',', token);



341    if (strlen(token) > 0) {



342        mlGpsData.cog = FLOAT_TO_UINT16_1E2((unsigned short) atof(token));



343    }



344



345    // 9.- UTC Date



346    // ddmmyy



347    myTokenizer(NULL, ',', token);



348    if (strlen(token) > 5) {



349        // get day



350//        tmp[0] = token[0];



351//        tmp[1] = token[1];



352//        mlGpsDateTime.day = (unsigned char) atoi(tmp);



353//        // get month



354//        tmp[0] = token[2];



355//        tmp[1] = token[3];



356//        mlGpsDateTime.month = (unsigned char) atoi(tmp);



357//        // get year



358//        tmp[0] = token[4];



359//        tmp[1] = token[5];



360//        mlGpsDateTime.year = (unsigned char) atoi(tmp);



361    }



362}

466// GPS checksum code based on 



467// http://www.codeproject.com/KB/mobile/WritingGPSApplications2.aspx



468// original code in C# written by Jon Person, author of "GPS.NET" (www.gpsdotnet.com)



469



470unsigned char getChecksum(unsigned char* sentence, unsigned char size) {



471



472    // Loop through all chars to get a checksum



473    unsigned char checkSum = 0;



474    unsigned char i;



475    for (i = 0; i < size; i++) {



476        if (sentence[i] == DOLLAR) {



477            // Ignore the dollar sign



478        } else if (sentence[i] == STAR) {



479            // Stop processing before the asterisk



480            break;



481        } else {



482            // Is this the first value for the checksum?



483            if (i == 0) {



484                // Yes. Set the checksum to the value



485                checkSum = sentence[i];



486            } else {



487                // No. XOR the checksum with this character's value



488                checkSum ^= sentence[i];



489            }



490        }



491    }



492    // Return the checksum 



493    return checkSum;



494}
