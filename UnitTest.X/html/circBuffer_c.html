   54   // this Function returns a pointer to a new Circular Buffer of size pm_size
   55   
   56   #if __IN_DSPIC__
   57   	void newCircBuffer (CBRef cB){
   58   
   59   		// initialize to zero
   60   		int i;
   61   		for (i=0; i<BSIZE; i++){
   62   			cB->buffer[i] = 0;
   63   		}
   64   
   65   		// initialize the data members
   66   		cB->head = 0;
   67   		cB->tail = 0;
   68   		cB->size = BSIZE;
   69   		cB->overflowCount = 0;
   70   
   71   	}
   72   
   73   //this function frees the Circular Buffer CB Ref
   74   	void freeCircBuffer (CBRef* cB){
   75   		// if it is already null, nothing to free
   76   		if (cB == NULL || *cB == NULL) {return;}
   77   
   78   		// free and nil the pointer
   79   		//free(*cB);
   80   		*cB = NULL;
   81   	}
   82   
   83   
   84   #else
   85   	CBRef newCircBuffer (int pm_size){
   86   		// create the circular buffer pointer
   87   		CBRef cB;
   88   
   89   		// allocate memory for it
   90   		cB = (CBRef) malloc(sizeof(CircBuffer));
   91   
   92   		// allocate memory for the buffer
   93   		cB->buffer = (unsigned char *)calloc(pm_size, sizeof(unsigned char));
   94   
   95   		// initialize the data members
   96   		cB->head = 0;
   97   		cB->tail = 0;
   98   		cB->size = pm_size;
   99   		cB->overflowCount = 0;
  100   
  101   		// return the buffer pointer
  102   		return (cB);
  103   
  104   	}

  108   	// this function frees the Circular Buffer CB Ref
  109   	void freeCircBuffer (CBRef* cB){
  110   		// if it is already null, nothing to free
  111   		if (cB == NULL || *cB == NULL) {return;}
  112   
  113   		// free the buffer
  114   		free((*cB)->buffer);
  115   
  116   		// free and nil the pointer
  117   		free(*cB);
  118   		*cB = NULL;
  119   	}

  125   // Accesor Methods
  126   // ===============
  127   
  128   // returns the amount of unread bytes in the circular buffer
  129   unsigned int getLength (CBRef cB){
  130   	// if the circular buffer is not null
  131   	if (cB != NULL){
  132   		if (cB->head <= cB->tail){
  133   			return (cB->tail-cB->head);
  134   		} else{
  135   			return (cB->size + cB->tail - cB->head);
  136   		}
  137   	}
  138   	else{
  139   		return 0;
  140   	}
  141   
  142   
  143   }
  144   
  145   // returns the actual index of the head
  146   int readHead (CBRef cB){
  147   	// if the circular buffer is not null
  148   	if (cB != NULL){
  149   		return (cB->head);
  150   	}
  151   	else{
  152   		return 0;
  153   	}
  154   
  155   }
  156   
  157   // returns the actual index of the tail
  158   int readTail (CBRef cB){
  159   	// if the circular buffer is not null
  160   	if (cB != NULL){
  161   		return (cB->tail);
  162   	}
  163   	else{
  164   		return 0;
  165   	}
  166   
  167   }
  168   
  169   // returns the byte (actual value) that the head points to. this
  170   // does not mark the byte as read, so succesive calls to peak will
  171   // always return the same value
  172   unsigned char peak(CBRef cB){
  173   	// if the circular buffer is not null
  174   	if (cB != NULL)
  175   	{
  176   		// if there are bytes in the buffer
  177   		if (getLength(cB) > 0){
  178   			return cB->buffer[cB->head];
  179   		}
  180   	}
  181   	return 0;
  182   }

  185   // Manipulation Procedures
  186   // ======================
  187   // returns the front of the circular buffer and marks the byte as read
  188   unsigned char readFront (CBRef cB){
  189   	// if the circular buffer is not null
  190   	if (cB != NULL)
  191   	{
  192   		char retVal;
  193   		// if there are bytes in the buffer
  194   		if (getLength(cB) > 0){
  195   			retVal = cB->buffer[cB->head];
  196   			cB->head = cB->head < (cB->size -1)? cB->head+1: 0;
  197   			return retVal;
  198   		}
  199   		return 128;
  200   	}
  201   	return 254;
  202   }

  204   
  205   // writes one byte at the end of the circular buffer,
  206   // increments overflow count if overflow occurs
  207   unsigned char writeBack (CBRef cB, unsigned char data){
  208   	// if the circular buffer is not null
  209           unsigned char outvalue = 1;
  210   	if (cB != NULL){
  211   		if (getLength (cB) == (cB->size -1)){
  212   			cB->overflowCount ++;
  213   			//return 1;
  214   		} else {
  215   			cB->buffer[cB->tail] = data;
  216   			cB->tail = cB->tail < (cB->size -1)? cB->tail+1: 0;
  217   			outvalue = 0;
  218   		}
  219   		//return 0;
  220   	}
  221   	return outvalue;
  222   }
  223   
  224   // empties the circular buffer. It does not change the size. use with caution!!
  225   void makeEmpty(CBRef cB){
  226   	if (cB != NULL){
  227   		int i;
  228   		for(i = 0; i < cB->size; ++i)
  229   		{
  230   			cB->buffer[i]= 0;
  231   		}
  232   		cB->head = 0;
  233   		cB->tail = 0;
  234   		cB->overflowCount = 0;
  235   	}
  236   }
