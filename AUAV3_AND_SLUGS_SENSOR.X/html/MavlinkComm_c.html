32void __attribute__((__interrupt__,__auto_psv__)) _U1RXInterrupt(void)



33{



34  //



35        _U1RXIF = 0;



36    // Read the buffer while it has data



37    // and add it to the circular buffer



38    while (U1STAbits.URXDA == 1) {



39        writeBack(uartMavlinkInBuffer, (unsigned char) U1RXREG);



40    }



41



42    // If there was an overun error clear it and continue



43    if (U1STAbits.OERR == 1) {



44        _U1RXIF = 0;



45        U1STAbits.OERR = 0;



46    }



47



48    // clear the interrupt



49    IFS0bits.U1RXIF = 0;



50

83void protDecodeMavlink(void) {



84

111                    // Handle message



112            switch (msg.msgid) {

116                    mlPending.heartbeatAge = 0;



117                    break;



118//AM DBG



119                case MAVLINK_MSG_ID_MISSION_COUNT:

140                case MAVLINK_MSG_ID_MISSION_REQUEST_LIST:

145                        mlPending.wpTransaction = 1;

147                        // change the state

150

157



158                case MAVLINK_MSG_ID_MISSION_REQUEST:

164



165                        // reset the rest of the state machine

167                        mlPending.wpTimeOut = 0;



168                    } else {



169                        // TODO: put here a report for a single WP, i.e. not inside a transaction



170                    }



171                    break;



172



173                case MAVLINK_MSG_ID_MISSION_ACK:



174                    mavlink_msg_mission_ack_decode(&msg, &mlWpAck);



175



176                    if (mlPending.wpTransaction) {



177                        // End the transaction



178                        mlPending.wpTransaction = 0;



179



180                        // change the state



181                        mlPending.wpProtState = WP_PROT_IDLE;



182



183                        // reset the rest of the state machine



184                        mlPending.wpCurrentWpInTransaction = 0;



185                        mlPending.wpTimeOut = 0;



186



187                        // send current waypoint index



188                        mlPending.wpSendCurrent = TRUE;



189                    }



190



191                    break;



192



193                case MAVLINK_MSG_ID_MISSION_ITEM:



194                    writeSuccess = SUCCESS;



195                    mavlink_msg_mission_item_decode(&msg, &mlSingleWp);



196



197                    if (mlPending.wpTransaction && (mlPending.wpProtState == WP_PROT_RX_WP)) {



198                        mlPending.wpProtState = WP_PROT_GETTING_WP_IDLE;



199



200                    }



201

203

205                    mlWpValues.lon[indx] = mlSingleWp.y;



206                    mlWpValues.alt[indx] = mlSingleWp.z;



207



208                    mlWpValues.type[indx] = mlSingleWp.command;



209



210                    mlWpValues.orbit[indx] = (uint16_t) mlSingleWp.param3;



211/*



212                    // Record the data in EEPROM



213                    writeSuccess = storeWaypointInEeprom(&mlSingleWp);



214

216                    // if the write was not successful

218                        mlPending.wpAck++;



219



220                        mlWpAck.target_component = MAV_COMP_ID_MISSIONPLANNER;



221                        mlWpAck.type = MAV_MISSION_ERROR;



222



223                    }



224*/



225                    break;



226



227                case MAVLINK_MSG_ID_MISSION_CLEAR_ALL:



228



229                    writeSuccess = SUCCESS;



230



231                    // clear the WP values in memory;



232                    memset(&mlWpValues, 0, sizeof (mavlink_mission_item_values_t));



233/*



234                    writeSuccess = clearWaypointsFrom(0);



235



236                    // Set the flag of Aknowledge fail



237                    // if the write was unsuccessful



238                    if (writeSuccess != SUCCESS) {



239                        mlPending.statustext++;



240



241                        mlStatustext.severity = MAV_SEVERITY_ERROR;



242                        strncpy(mlStatustext.text, "Failed to clear waypoints from EEPROM.", 49);



243



244                    }



245  */



246



247                    // Update the waypoint count



248                    mlWpValues.wpCount = 0;



249



250                    // Set the state machine ready to send the WP akn



251                    mlPending.wpCurrentWpInTransaction = 0;



252                    mlPending.wpTotalWps = 0;



253                    mlPending.wpTransaction = 1;



254                    mlPending.wpProtState = WP_PROT_GETTING_WP_IDLE;



255



256                    break;



257



258                case MAVLINK_MSG_ID_SET_GPS_GLOBAL_ORIGIN:



259                    writeSuccess = SUCCESS;



260



261                    memset(&mlSingleWp, 0, sizeof (mavlink_mission_item_t));



262



263                    mavlink_msg_set_gps_global_origin_decode(&msg, &mlGSLocation);



264



265                    mlSingleWp.x = (float) (mlGSLocation.latitude);



266                    mlSingleWp.y = (float) (mlGSLocation.longitude);



267                    mlSingleWp.z = (float) (mlGSLocation.altitude);



268



269                    indx = (uint8_t) MAX_NUM_WPS - 1;



270



271                    mlWpValues.lat[indx] = mlSingleWp.x;



272                    mlWpValues.lon[indx] = mlSingleWp.y;



273                    mlWpValues.alt[indx] = mlSingleWp.z;



274                    mlWpValues.type[indx] = MAV_CMD_NAV_LAND;



275                    mlWpValues.orbit[indx] = 0;



276



277                    // Record the data in EEPROM



278                    /*



279                    writeSuccess = storeWaypointInEeprom(&mlSingleWp);

284                        mlStatustext.severity = MAV_SEVERITY_ERROR;

287                    else {

289                        mlPending.statustext++;

292                        strncpy(mlStatustext.text, "Control DSC GPS origin set.", 49);



293                    }



294                     */



295                    break;



296



297//AM DBG



298                case MAVLINK_MSG_ID_PARAM_REQUEST_LIST:



299                    mlPending.piTransaction = 1;



300                    mlPending.piProtState = PI_SEND_ALL_PARAM;



301                    mlPending.piCurrentParamInTransaction = 0;



302                    break;



303



304                case MAVLINK_MSG_ID_PARAM_REQUEST_READ:



305                    // If it was in the middle of a list transmission or there is already a param enqueued



306                    mlPending.piTransaction = 1;



307                    switch (mlPending.piProtState) {



308                        case PI_IDLE:



309                            mlPending.piBackToList = 0; // no need to go back



310                            mlPending.piQIdx = -1; // no Index



311                            mlPending.piCurrentParamInTransaction = mavlink_msg_param_request_read_get_param_index(&msg); // assign directly



312                            mlPending.piProtState = PI_SEND_ONE_PARAM;



313                            break;



314



315                        case PI_SEND_ALL_PARAM:



316                            mlPending.piBackToList = 1; // mark to go back



317                            mlPending.piQIdx++; // done like this because when empty index = -1



318                            mlPending.piQueue[mlPending.piQIdx] = mavlink_msg_param_request_read_get_param_index(&msg); // put in in queue



319                            mlPending.piProtState = PI_SEND_ONE_PARAM;



320                            break;



321



322                        case PI_SEND_ONE_PARAM:



323                            if (mlPending.piBackToList) {



324                                mlPending.piQIdx++; // done like this because when empty index = -1



325                                mlPending.piQueue[mlPending.piQIdx] = mavlink_msg_param_request_read_get_param_index(&msg); // put in in queue



326                            }



327                            mlPending.piProtState = PI_SEND_ONE_PARAM;



328                            break;



329                    }



330                    break;



331



332                case MAVLINK_MSG_ID_PARAM_SET:



333                    mavlink_msg_param_set_decode(&msg, &set);



334



335                    if ((uint8_t) set.target_system == (uint8_t) SYSTEMID &&



336                        (uint8_t) set.target_component == (uint8_t) COMPID) {



337



338



339                        char* key = (char*) set.param_id;



340                        uint8_t i, j;



341                        uint8_t match;



342                        for (i = 0; i < PAR_PARAM_COUNT; i++) {



343                            match = 1;



344                            for (j = 0; j < PARAM_NAME_LENGTH; j++) {



345                                // Compare



346                                if (((char) (mlParamInterface.param_name[i][j]))



347                                    != (char) (key[j])) {



348                                    match = 0;



349                                } // if



350



351                                // End matching if null termination is reached



352                                if (((char) mlParamInterface.param_name[i][j]) == '\0') {



353                                    break;



354                                } // if



355                            }// for j



356



357                            // Check if matched



358                            if (match) {



359                                //sw_debug = 1;



360                                // Only write and emit changes if there is actually a difference



361                                // AND only write if new value is NOT "not-a-number"



362                                // AND is NOT infinity



363



364                                if (isFinite(set.param_value)) {



365



366                                    mlParamInterface.param[i] = set.param_value;



367



368                                    // Report back new value



369                                    mlPending.piBackToList = 0; // no need to go back



370                                    mlPending.piQIdx = -1; // no Index



371                                    mlPending.piCurrentParamInTransaction = i; // assign directly



372                                    mlPending.piProtState = PI_SEND_ONE_PARAM;



373                                    mlPending.piTransaction = 1;



374



375                                } // if different and not nan and not inf



376                            } // if match



377                        }// for i



378                    } // if addressed to this



379                    break;



380              default:



381                break;



382            }



383       }



384    }



385}

599              mlPending.piProtState = PI_IDLE;



600              mlPending.piTransaction = 0;



601          } else { // you will need to go back to send all



602



603              if (mlPending.piQIdx < 0) { // if you've sent all the requests, then go back to list



604                  mlPending.piProtState = PI_SEND_ALL_PARAM;



605                  mlPending.piBackToList = 0;



606



607              } else { // send the requests



608                  mavlink_msg_param_value_pack(system_id,



609                      component_id,



610                      &msg,



611                      mlParamInterface.param_name[mlPending.piQueue[mlPending.piQIdx]],



612                      mlParamInterface.param[mlPending.piQueue[mlPending.piQIdx]],



613                      MAV_PARAM_TYPE_REAL32,



614                      PAR_PARAM_COUNT,



615                      mlPending.piQueue[mlPending.piQIdx]);



616



617                  // Copy the message to the send buffer



618                  bytes2Send += mavlink_msg_to_send_buffer(UartOutBuff, &msg);



619



620                  // decrement the queue index



621                  mlPending.piQIdx--;



622              } // QIdx < 0



623          }// !backToList



624    }//switch



625  }//if (mlPending.piTransaction)



626  return(bytes2Send);



627}

780extern MCHP_UART1_TxStr MCHP_UART1_Tx;

782void TxN_Data_OverU1(uint16_t N){



783  uint16_T i;



784  for (i = 0U; i < N; i++) {



785    uint16_T Tmp;



786    Tmp = ~(MCHP_UART1_Tx.tail - MCHP_UART1_Tx.head);



787    Tmp = Tmp & (Tx_BUFF_SIZE_Uart1 - 1);/* Modulo Buffer Size */



788    if (Tmp != 0) {



789      MCHP_UART1_Tx.buffer[MCHP_UART1_Tx.tail] = UartOutBuff[i];



790      MCHP_UART1_Tx.tail = (MCHP_UART1_Tx.tail + 1) & (Tx_BUFF_SIZE_Uart1 - 1);



791      Tmp--;



792    }



793  }



794  _U1TXIF = U1STAbits.TRMT;



795}
