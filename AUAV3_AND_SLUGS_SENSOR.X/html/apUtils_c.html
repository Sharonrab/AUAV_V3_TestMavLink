130void hugeDelay(void) {



131    int i, j;



132    for (i = 0; i < 750; i += 1) {



133        for (j = 0; j < 32700; j += 1) {



134            Nop();



135        }



136    }



137}



138



139void smallDelay(void) {



140    int i;



141    for (i = 0; i < 5000; i += 1) {



142        Nop();



143    }



144}



145



146



147void tinyDelay(void) {



148    int i;



149    for (i = 0; i < 50; i += 1) {



150        Nop();



151    }



152}
