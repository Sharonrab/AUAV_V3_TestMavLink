    3   // This is code implements a fully interrupt driven UART reader to
    4   // be used in the UCSC Autopilot project. It makes use of the
    5   // circular buffer data structure circBuffer.c. It has been
    6   // written to be implemented in Simulink. It configures UART 1
    7   // at a predefined baud rate, then initializes a circular buffer,
    8   // configures the interrupt and starts the service.

   10   // how many new bytes were read, and byte m indicates how many remain

   29   void uartBufferInit(void) {
   30   
   31        /* Configure Pins as Analog or Digital */
   32       ANSELE = 0xFF83;
   33   
   34       /* Configure Remappables Pins */
   35       RPINR28 = 0x56;
   36       /* Configure UART4 Rx Interruption */
   37       _U4RXIP = 1;                         /* Rx Interrupt priority set to 1 */
   38       _U4RXIF = 0;
   39       _U4RXIE = 1;                         /* Enable Interrupt */
   40       uartBuffer = (struct CircBuffer*) &com4Buffer;
   41       newCircBuffer(uartBuffer);
   42   }

   44   // Interrupt service routine for U4 GPS port
   45   void __attribute__((__interrupt__, no_auto_psv)) _U4RXInterrupt(void) {
   46         _U4RXIF = 0;
   47       // Read the buffer while it has data
   48       // and add it to the circular buffer
   49       while (U4STAbits.URXDA == 1) {
   50           writeBack(uartBuffer, (unsigned char) U4RXREG);
   51       }
   52   
   53       // If there was an overun error clear it and continue
   54       if (U4STAbits.OERR == 1) {
   55           _U4RXIF = 0;
   56           U4STAbits.OERR = 0;
   57       }
   58   
   59       // clear the interrupt
   60       IFS5bits.U4RXIF = 0;
   61   }
