595      MCHP_I2C2_State++;



596      break;



597



598     case 101:



599      I2C2CONbits.RCEN = 1;            /* I2C: Receive  */



600      MCHP_I2C2_State++;



601      break;



602



603     case 102:



604      I2C24_Buff8[4] = I2C2RCV;        /* I2C : Read data */



605      I2C2CONbits.ACKDT = 0;           /* set to ACK */



606      I2C2CONbits.ACKEN = 1;           /* Start Acknowledge sequence */



607      MCHP_I2C2_State++;



608      break;



609



610     case 103:



611      I2C2CONbits.RCEN = 1;            /* I2C: Receive  */



612      MCHP_I2C2_State++;



613      break;



614



615     case 104:



616      I2C24_Buff8[5] = I2C2RCV;        /* I2C : Read data */



617      I2C2CONbits.ACKDT = 1;           /* set to NACK */



618      I2C2CONbits.ACKEN = 1;           /* Start Acknowledge sequence */



619      MCHP_I2C2_State++;



620      break;



621



622     case 105:



623      I2C2CONbits.PEN = 1;             /* I2C: STOP */



624      MCHP_I2C2_State++;



625      break;



626



627     case 106:



628      MCHP_I2C24_Request = 0;          /* This I2C sequence could be Re-Enabled */



629      MCHP_I2C2_State = 0;



630      Continue = (MCHP_I2C2_Queue.head != MCHP_I2C2_Queue.tail);/* One next sequence queued, start it now */



631      break;



632



633     case 107:                         /* Start a new I2C Sequence */



634      I2C2CONbits.SEN = 1;             /* I2C: START  */



635      MCHP_I2C2_State++;



636      break;



637



638     case 108:



639      I2C2TRN = 0x3C;                  /* Send I2C Address : A7 A6 A5 A4 A3 A2 A1 A0 RW */



640      MCHP_I2C2_State++;



641      break;



642



643     case 109:



644      I2C2TRN = 0x00;                  /* I2C : Write data */



645      MCHP_I2C2_State++;



646      break;



647



648     case 110:



649      I2C2TRN = 0x78;                  /* I2C : Write data */



650      MCHP_I2C2_State++;



651      break;



652



653     case 111:



654      I2C2TRN = 0x00;                  /* I2C : Write data */



655      MCHP_I2C2_State++;



656      break;



657



658     case 112:



659      I2C2TRN = 0x00;                  /* I2C : Write data */



660      MCHP_I2C2_State++;



661      break;



662



663     case 113:



664      I2C2CONbits.PEN = 1;             /* I2C: STOP */



665      MCHP_I2C2_State++;



666      break;



667



668     case 114:



669      MCHP_I2C25_Request = 0;          /* This I2C sequence could be Re-Enabled */



670      MCHP_I2C2_State = 0;



671      Continue = (MCHP_I2C2_Queue.head != MCHP_I2C2_Queue.tail);/* One next sequence queued, start it now */



672      break;



673



674     default:                          /* Sequence finished */



675      MCHP_I2C2_State = 0;             /* Should never happend */



676      break;



677    }                                  /* End of switch case sequence*/



678  } while (Continue == 1);             /* end do-While */



679



680  asm("DEC _mcuFlagRecursion");        /* ensure --mcuFlagRecursion is atomic */



681  if (mcuFlagRecursion == 0) {



682    T2CONbits.TON = 0;



683  }



684}                                      /* Enf of interrupt */
