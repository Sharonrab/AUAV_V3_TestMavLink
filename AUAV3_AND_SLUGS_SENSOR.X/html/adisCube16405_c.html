143void getCubeData (short * cubeData) {



144	// Do the last reading



145	write2Cube(R_GYROX);



146	



147	cubeBuffer.gx[cubeBuffer.sampleCount] = convert14BitToShort(write2Cube(R_GYROY));



148	cubeBuffer.gy[cubeBuffer.sampleCount] = convert14BitToShort(write2Cube(R_GYROZ));



149	cubeBuffer.gz[cubeBuffer.sampleCount] = convert14BitToShort(write2Cube(R_ACCELX));



150	cubeBuffer.ax[cubeBuffer.sampleCount] =convert14BitToShort(write2Cube(R_ACCELY));



151	cubeBuffer.ay[cubeBuffer.sampleCount] =convert14BitToShort(write2Cube(R_ACCELZ));



152	cubeBuffer.az[cubeBuffer.sampleCount] =convert14BitToShort(write2Cube(R_MAGNX)); 



153	cubeBuffer.mx[cubeBuffer.sampleCount] =convert14BitToShort(write2Cube(R_MAGNY)); 



154	cubeBuffer.my[cubeBuffer.sampleCount] =convert14BitToShort(write2Cube(R_MAGNZ)); 



155	cubeBuffer.mz[cubeBuffer.sampleCount] =convert14BitToShort(write2Cube(R_STATUS)); // dummy write



156	cubeBuffer.sampleCount++;



157	



158	cubeData[9]= cubeBuffer.sampleCount;		// se incorporaron los magnetometros



159	// Update the state structure with the collected values



160	updateCubeData();



161	



162	// Return the averaged values



163	cubeData[0] = mlRawImuData.xgyro;



164	cubeData[1] = mlRawImuData.ygyro;



165	cubeData[2] = mlRawImuData.zgyro;



166	cubeData[3] = mlRawImuData.xacc;



167	cubeData[4] = mlRawImuData.yacc;



168	cubeData[5] = mlRawImuData.zacc;



169	cubeData[6] = mlRawImuData.xmag;



170	cubeData[7] = mlRawImuData.ymag;



171	cubeData[8] = mlRawImuData.zmag;



172	



173	// Turn on the timer for the first MAX_CUBE_READ readings



174	T5CONbits.TON				= 1;



175}

177void updateCubeData(void){



178		



179	// Read the buffered measurements



180	mlRawImuData.xgyro = averageData(cubeBuffer.gx, cubeBuffer.sampleCount);



181	mlRawImuData.ygyro = averageData(cubeBuffer.gy, cubeBuffer.sampleCount);



182	mlRawImuData.zgyro = averageData(cubeBuffer.gz, cubeBuffer.sampleCount);



183



184	mlRawImuData.xacc = averageData(cubeBuffer.ax, cubeBuffer.sampleCount);



185	mlRawImuData.yacc = averageData(cubeBuffer.ay, cubeBuffer.sampleCount);



186	mlRawImuData.zacc = averageData(cubeBuffer.az, cubeBuffer.sampleCount);



187	



188	mlRawImuData.xmag = averageData(cubeBuffer.mx, cubeBuffer.sampleCount);



189	mlRawImuData.ymag = averageData(cubeBuffer.my, cubeBuffer.sampleCount);



190	mlRawImuData.zmag = averageData(cubeBuffer.mz, cubeBuffer.sampleCount);



191	



192	// Clear the count for the next reading



193	cubeBuffer.sampleCount = 0;



194 



195}

197int16_t averageData(int16_t* theData, uint8_t count){			



198	char i;



199	short value = 0;



200	for (i=0; i< count; i++){



201		value+= theData[(int)i];



202	}



203	



204	return (count>0)? ((int16_t)(value/count)) : 0;



205}

245// interrupcion del timer5



246void __attribute__ ((interrupt, no_auto_psv)) _T5Interrupt(void)



247{



248	// Do the reading



249	// lee datos del gyro



250	write2Cube(R_GYROX);



251	cubeBuffer.gx[cubeBuffer.sampleCount] = convert14BitToShort(write2Cube(R_GYROY));



252	cubeBuffer.gy[cubeBuffer.sampleCount] = convert14BitToShort(write2Cube(R_GYROZ));



253	//lee datos del acelerometro



254	cubeBuffer.gz[cubeBuffer.sampleCount] = convert14BitToShort(write2Cube(R_ACCELX));



255	cubeBuffer.ax[cubeBuffer.sampleCount] =convert14BitToShort(write2Cube(R_ACCELY));



256	cubeBuffer.ay[cubeBuffer.sampleCount] =convert14BitToShort(write2Cube(R_ACCELZ));



257	// lee datos del magnetometro



258	cubeBuffer.az[cubeBuffer.sampleCount] =convert14BitToShort(write2Cube(R_MAGNX)); 



259	cubeBuffer.mx[cubeBuffer.sampleCount] =convert14BitToShort(write2Cube(R_MAGNY)); 



260	cubeBuffer.my[cubeBuffer.sampleCount] =convert14BitToShort(write2Cube(R_MAGNZ)); 



261	cubeBuffer.mz[cubeBuffer.sampleCount] =convert14BitToShort(write2Cube(R_STATUS)); // dummy write



262	cubeBuffer.sampleCount++;



263		



264	// If this is the MAX_CUBE_READ read value turn the timer off



265	if(MAX_CUBE_READ <= cubeBuffer.sampleCount){



266		T5CONbits.TON = 0;



267	}



268	



269	IFS1bits.T5IF = 0;		// clear the interrupt flag



270	TMR5 = 0;							// Start the count again



271}

273short convert14BitToShort (short wordData) {



274	return (wordData & BITTEST_14)? (wordData | BITEXTEND_14) : (wordData & BITMASK_14);



275}



276



277short convert12BitToShort (short wordData) {



278	return (wordData & BITTEST_12)? (wordData | BITEXTEND_12) : (wordData & BITMASK_12);



279}
