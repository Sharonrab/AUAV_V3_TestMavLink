  130   void hugeDelay(void) {
  131       int i, j;
  132       for (i = 0; i < 750; i += 1) {
  133           for (j = 0; j < 32700; j += 1) {
  134               Nop();
  135           }
  136       }
  137   }

  164   void printToUart2 (const char *fmt, ...){
  165   	va_list ap;
  166   	char buf [300];
  167   	
  168   	va_start(ap, fmt);
  169   	vsprintf(buf, fmt, ap);
  170   	va_end (ap);
  171   	putsUART2((unsigned int*)buf);
  172            while (BusyUART2());
  173   }
