4/* Implement SPI 1 Interrupts */



5void __attribute__((__interrupt__,__shadow__,__auto_psv__)) _SPI1Interrupt(void)



6  /* SPI1 ISR */



7{



8  boolean_T Continue;



9  _SPI1IF = 0;



10  asm("INC _mcuFlagRecursion");        /* ensure atomic	mcuFlagRecursion++; */



11  T2CONbits.TON = 1;



12



13  /* Declaration of Variables */



14  uint16_T tmp;



15  do {



16    Continue = 0;                      /* By default, execute one sequence only */



17    switch (MCHP_SPI1_State)



18    {



19     case 0:                           /* Idle */



20      /* Check if the next sequence is already in the queue. If so, start it */



21      Continue = (MCHP_SPI1_Queue.head != MCHP_SPI1_Queue.tail);/* One next sequence queued, start it now */



22      if (Continue) {



23        MCHP_SPI1_State = MCHP_SPI1_Queue.buffer[MCHP_SPI1_Queue.tail];



24        if (MCHP_SPI1_Queue.tail >= 2) /* There are 2 SPI1 blocks, max idx for queue is 2 */



25          MCHP_SPI1_Queue.tail = 0;



26        else



27          MCHP_SPI1_Queue.tail++;



28      }



29      break;



30



31     case 1:                           /* Start a new SPI Sequence */



32      /* Toggle pin */



33      LATEbits.LATE4 = 1;              /* switch RE4 to 1 */



34



35      /* Delay */



36      __delay32(700000);               /* Delay  */



37



38      /* Toggle pin */



39      LATEbits.LATE4 = 0;              /* switch RE4 to 0 */



40



41      /* Write only Sequence */



42      SPI1BUF = 0x6B80;



43



44      /* Toggle pin */



45      MCHP_SPI1_State++;



46      break;



47



48     case 2:



49      LATEbits.LATE4 = 1;              /* switch RE4 to 1 */



50



51      /* Delay */



52      __delay32(4200000);              /* Delay  */



53



54      /* Toggle pin */



55      LATEbits.LATE4 = 0;              /* switch RE4 to 0 */



56



57      /* Write only Sequence */



58      tmp = SPI1BUF;



59      SPI1BUF = 0x6B01;



60



61      /* Toggle pin */



62      MCHP_SPI1_State++;



63      break;



64



65     case 3:



66      LATEbits.LATE4 = 1;              /* switch RE4 to 1 */



67



68      /* Delay */



69      __delay32(700000);               /* Delay  */



70



71      /* Toggle pin */



72      LATEbits.LATE4 = 0;              /* switch RE4 to 0 */



73



74      /* Write only Sequence */



75      tmp = SPI1BUF;



76      SPI1BUF = 0x6A10;



77



78      /* Toggle pin */



79      MCHP_SPI1_State++;



80      break;



81



82     case 4:



83      LATEbits.LATE4 = 1;              /* switch RE4 to 1 */



84



85      /* Delay */



86      __delay32(700000);               /* Delay  */



87



88      /* Toggle pin */



89      LATEbits.LATE4 = 0;              /* switch RE4 to 0 */



90



91      /* Write only Sequence */



92      tmp = SPI1BUF;



93      SPI1BUF = 0x1904;



94



95      /* Toggle pin */



96      MCHP_SPI1_State++;



97      break;



98



99     case 5:



100      LATEbits.LATE4 = 1;              /* switch RE4 to 1 */



101



102      /* Delay */



103      __delay32(700000);               /* Delay  */



104



105      /* Toggle pin */



106      LATEbits.LATE4 = 0;              /* switch RE4 to 0 */



107



108      /* Write only Sequence */



109      tmp = SPI1BUF;



110      SPI1BUF = 0x1A03;



111



112      /* Toggle pin */



113      MCHP_SPI1_State++;



114      break;



115



116     case 6:



117      LATEbits.LATE4 = 1;              /* switch RE4 to 1 */



118



119      /* Delay */



120      __delay32(700000);               /* Delay  */



121



122      /* Toggle pin */



123      LATEbits.LATE4 = 0;              /* switch RE4 to 0 */



124



125      /* Write only Sequence */



126      tmp = SPI1BUF;



127      SPI1BUF = 0x1B08;



128



129      /* Toggle pin */



130      MCHP_SPI1_State++;



131      break;



132



133     case 7:



134      LATEbits.LATE4 = 1;              /* switch RE4 to 1 */



135      tmp = SPI1BUF;



136      MCHP_SPI12_Request = 0;          /* This SPI sequence could be Re-Enabled */



137      MCHP_SPI1_State = 0;             /* End of SPI Sequence. SPI Interrupt should be started for new sequence */



138      Continue = (MCHP_SPI1_Queue.head != MCHP_SPI1_Queue.tail);/* One next sequence queued, start it now */



139      break;



140



141     case 8:                           /* Start a new SPI Sequence */



142      /* Toggle pin */



143      LATEbits.LATE4 = 0;              /* switch RE4 to 0 */



144



145      /* Write only Sequence */



146      SPI1BUF = 0xBA00;



147



148      /* Read only Sequence */



149      SPI1BUF = 0x0000;



150      SPI1BUF = 0x0000;



151      SPI1BUF = 0x0000;



152



153      /* Read only Sequence */



154      SPI1BUF = 0x0000;



155



156      /* Read only Sequence */



157      SPI1BUF = 0x0000;



158      SPI1BUF = 0x0000;



159      SPI1BUF = 0x0000;



160      MCHP_SPI1_State++;



161      break;



162



163     case 9:



164      /* Toggle pin */



165      LATEbits.LATE4 = 1;              /* switch RE4 to 1 */



166      tmp = SPI1BUF;



167      SPI11_Buff16[0] = SPI1BUF;



168      SPI11_Buff16[1] = SPI1BUF;



169      SPI11_Buff16[2] = SPI1BUF;



170      SPI11_Buff16[3] = SPI1BUF;



171      SPI11_Buff16[4] = SPI1BUF;



172      SPI11_Buff16[5] = SPI1BUF;



173      SPI11_Buff16[6] = SPI1BUF;



174      MCHP_SPI11_Request = 0;          /* This SPI sequence could be Re-Enabled */



175      MCHP_SPI1_State = 0;             /* End of SPI Sequence. SPI Interrupt should be started for new sequence */



176      Continue = (MCHP_SPI1_Queue.head != MCHP_SPI1_Queue.tail);/* One next sequence queued, start it now */



177      break;



178



179     default:                          /* Sequence finished */



180      MCHP_SPI1_State = 0;             /* Should never happend */



181      break;



182    }                                  /* End of switch case sequence*/



183  } while (Continue == 1);             /* end do-While */



184



185  asm("DEC _mcuFlagRecursion");        /* ensure --mcuFlagRecursion is atomic */



186  if (mcuFlagRecursion == 0) {



187    T2CONbits.TON = 0;



188  }



189}                                      /* Enf of interrupt */
