25void gpsParse(uint8_t* dataStream) {



26



27    DECODER_STATUS static currentStatus = DECO_START;



28



29    uint16_t static payloadLen = 0;



30    uint16_t static bytesRead = 0;



31    uint8_t static completeMessage [MSIZE];



32    uint16_t static totalBytes = 0;



33    uint32_t static currentChecksum = 0;



34



35    uint8_t ii = 1;



36    uint8_t bytesReceived = dataStream[0];



37    tUnsignedShortToChar shTemp;



38



39    while (bytesReceived) {



40        switch (currentStatus) {



41



42            case DECO_START:



43                payloadLen = 0;



44                bytesRead = 0;



45                totalBytes = 0;



46                currentChecksum = 0;



47



48                while (dataStream[ii] != HEAD1) {



49                    ii++;



50                    bytesReceived--;



51



52                    if (bytesReceived == 0) {



53                        return;



54                    }



55                }



56                completeMessage[bytesRead++] = HEAD1;



57                ii++;



58                bytesReceived--;



59                currentStatus = DECO_RECEIVING_HEADER2;



60                break;



61



62            case DECO_RECEIVING_HEADER2:



63                if (dataStream[ii] != HEAD2) {



64                    currentStatus = DECO_START;



65



66                } else {



67                    ii++;



68                    bytesReceived--;



69                    completeMessage[bytesRead++] = HEAD2;



70                    currentStatus = DECO_RECEIVING_HEADER3;



71                }



72



73                break;



74



75            case DECO_RECEIVING_HEADER3:



76                if (dataStream[ii] != HEAD3) {



77                    currentStatus = DECO_START;



78                    bytesRead = 0;



79                } else {



80                    ii++;



81                    bytesReceived--;



82                    completeMessage[bytesRead++] = HEAD3;



83                    currentStatus = DECO_RECEIVING_HEADER;



84                }



85                break;



86



87            case DECO_RECEIVING_HEADER:



88                while ((bytesReceived > 0) && (bytesRead < HEAD_LEN)) {



89                    completeMessage[bytesRead++] = dataStream[ii++];



90                    bytesReceived--;



91                }



92



93                if (bytesRead == HEAD_LEN) {



94                    currentStatus = DECO_RECEIVING_PAYLOAD;



95



96                    payloadLen = charsToUint16(&(completeMessage[8]));



97



98                    totalBytes = payloadLen + HEAD_LEN + 4; // this includes the 4 CRC bytes



99                }



100                break;



101



102



103            case DECO_RECEIVING_PAYLOAD:



104                while ((bytesReceived > 0) && (bytesRead < totalBytes)) {



105                    completeMessage[bytesRead++] = dataStream[ii++];



106                    bytesReceived--;



107                }



108



109                if (bytesRead == totalBytes) {



110                    currentStatus = DECO_COMPUTING_CRC;



111



112                    currentChecksum = charsToUint32(&(completeMessage[bytesRead - 4]));



113                }



114                break;



115



116            case DECO_COMPUTING_CRC:



117                if (CalculateBlockCRC32(bytesRead - 4, completeMessage) == currentChecksum) {



118                    currentStatus = DECO_DECODING;



119                } else {



120                    mlNovatelStatus.csFails++;



121                    currentStatus = DECO_START;



122                }



123                break;



124



125            case DECO_DECODING:



126                shTemp.chData[0] = completeMessage[4];



127                shTemp.chData[1] = completeMessage[5];



128



129                updateStatus(completeMessage);



130



131                switch (shTemp.usData) {



132                    case BEST_POS_ID:



133                        updatePositionData(completeMessage);



134                        break;



135



136                    case BEST_VEL_ID:



137                        updateVelocityData(completeMessage);



138                        break;



139



140                    case TIME_ID:



141                        updateTimeData(completeMessage);



142                        break;



143



144                    default:



145



146                        break;



147                }



148



149                currentStatus = DECO_START;



150                break;



151        } // switch



152



153    }// while bytes received



154



155}

225/* --------------------------------------------------------------------------



226Calculate a CRC value to be used by CRC calculation functions.



227 -------------------------------------------------------------------------- */



228uint32_t CRC32Value(int32_t i) {



229    int32_t j;



230    uint32_t ulCRC;



231    ulCRC = i;



232    for (j = 8; j > 0; j--) {



233        if (ulCRC & 1) {



234            ulCRC = (ulCRC >> 1) ^ CRC32_POLYNOMIAL;



235        } else {



236            ulCRC >>= 1;



237        } // if



238    } // for



239    return ulCRC;



240}

242/* --------------------------------------------------------------------------



243Calculates the CRC-32 of a block of data all at once



244-------------------------------------------------------------------------- */



245uint32_t CalculateBlockCRC32(



246    uint32_t ulCount, /* Number of bytes in the data block */



247    uint8_t *ucBuffer) { /* Data block */



248



249    uint32_t ulTemp1;



250    uint32_t ulTemp2;



251    uint32_t ulCRC = 0;



252    while (ulCount-- != 0) {



253        ulTemp1 = (ulCRC >> 8) & 0x00FFFFFFL;



254        ulTemp2 = CRC32Value(((int32_t) ulCRC ^ *ucBuffer++) & 0xff);



255        ulCRC = ulTemp1 ^ ulTemp2;



256    }



257    return ( ulCRC);



258}
